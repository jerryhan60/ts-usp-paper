\begin{abstract}
\noindent
\emph{Universal Sequence Preconditioning} (USP)~\citep{marsden2024universal} is a framework for online prediction in marginally stable linear dynamical systems (LDS) in which convolving observations with a fixed monic Chebyshev polynomial compresses the effective spectral radius of the hidden dynamics, yielding the first sublinear, hidden-dimension-independent regret bounds for this class. This paper empirically investigates whether the USP idea transfers to a setting of patch-based transformer foundation models for time-series forecasting, which the original paper does not cover. The setting differs from USP in three ways: the forecaster is a transformer trained from scratch on a large time-series corpus, not an online learner; the data may be far richer than a finite-dimensional LDS; and the task is multi-step forecasting rather than single-step prediction. To accommodate these differences we propose \textbf{\method{}}. We precondition our input and concatenate it as an auxiliary input \emph{channel} alongside the raw target, sidestepping the catastrophic error accumulation that direct target preconditioning would induce under multi-step rollout. Only the per-patch input projection is widened ($+12$K parameters, $0.11\%$ of Moirai-2 Small). Empirically, on the GIFT-Eval benchmark ($97$ configurations, $5$ seeds, $485$ paired comparisons) our method achieves $\Delta\mathrm{MASE}^* = -2.9\%$ over a from-scratch Moirai-2 Small baseline, winning $329/485$ paired comparisons; capacity-matched controls (duplicate channel, zero channel) confirm that the specific polynomial content drives the gain. At $100$K training steps the improvement grows to $-3.9\%$, and the gain remains $\sim 5\%$ across context lengths $512$--$4{,}000$. Cross-benchmark validation on FEV-Bench ($-2.9\%$ MASE, $72/100$ wins) confirms the finding independently.
\end{abstract}

\subsection{Passive Prediction in Linear Dynamical Systems}
\label{sec:passive-lds}

A partially observed linear time-invariant system has hidden state $x_t \in \R^{d_x}$, observed input $u_t \in \R^{d_u}$, and output $y_t \in \R^{d_y}$:
\begin{equation}
    x_{t+1} = A x_t + B u_t + w_t,
    \qquad y_t = C x_t + v_t,
    \qquad t \in \Z_{\geq 0},
    \label{eq:lds}
\end{equation}
with initial state $x_0 \in \R^{d_x}$. Here $A, B, C$ are unknown system matrices, while $w_t$ and $v_t$ denote process and observation noise. In time-series forecasting there may be no exogenous input; this is recovered by setting $u_t \equiv 0$ or by treating past observations as the relevant input history.

The prediction problem is passive. Unlike control, the learner does not choose $u_t$ to influence the state. At each time $t$, the learner observes the past history and predicts $\hat y_t$; after observing $y_t$, it suffers loss $\ell_t(\hat y_t, y_t)$ (for example a squared error). For a predictor class $\Pi$, regret is
\begin{equation}
    \mathrm{Regret}_T(\mathcal{A}, \Pi) \;:=\; \sum_{t=1}^{T} \ell_t(\hat y_t, y_t) - \min_{\pi \in \Pi} \sum_{t=1}^{T} \ell_t(\hat y_t^{\pi}, y_t).
    \label{eq:prediction-regret}
\end{equation}
\begin{definition}[Sublinear regret]
\label{def:sublinear-regret}
An online algorithm $\mathcal{A}$ achieves \emph{sublinear regret} with respect to a comparator class $\Pi$ if $\mathrm{Regret}_T(\mathcal{A}, \Pi) = o(T)$, or equivalently,
\[
    \frac{1}{T} \mathrm{Regret}_T(\mathcal{A}, \Pi) \;\xrightarrow{T \to \infty}\; 0.
\]
That is, the learner's average per-round loss converges to that of the best predictor in the reference class. This is the sense in which the online-learning results below should be read.
\end{definition}

\subsection{Stability and Finite Memory}
\label{sec:stability-memory}

The \emph{spectral radius} of a square matrix is $\rho(A) := \max\{\abs{\lambda} : \lambda \in \mathrm{spec}(A)\}$. For a time-invariant linear system, $\rho(A) < 1$ is equivalent to $A^t \to 0$ as $t \to \infty$. For finite-time learning bounds one usually needs the quantitative version
\begin{equation}
    \norm{A^t} \;\leq\; M \rho^t \qquad \text{for all } t \geq 0, \qquad \rho \in (0, 1),
    \label{eq:quantitative-stability}
\end{equation}
which says that the influence of the remote past decays geometrically. The following truncation lemma is the precise version of the common statement that strictly stable systems have finite effective memory.

\begin{lemma}[Stable LDSs are approximated by finite-memory predictors; cf.\ {\citealp[Lemma~9.6]{hazan2025onlinecontrol}}]
\label{lem:stable-truncation}
Consider the noiseless input-output LDS $x_{t+1} = A x_t + B u_t$, $y_t = C x_t$. Suppose $\norm{A^t} \leq M \rho^t$, $\norm{B} \leq \beta$, $\norm{C} \leq \gamma$, $\norm{u_t} \leq U$ with $\rho \in (0, 1)$. Let $\pi_{A,B,C,x_0}$ denote the exact open-loop LDS predictor and let $\pi_{A,B,C,h}$ be the length-$h$ truncated predictor $\hat y_t^{(h)} = \sum_{i=1}^{h} CA^{i-1}B u_{t-i}$. Then
\begin{equation}
    \sum_{t=1}^{T} \norm{\hat y_t^{\pi_{A,B,C,x_0}} - \hat y_t^{(h)}} \;\leq\; T \cdot \frac{M \beta \gamma U}{1 - \rho} \rho^h + \sum_{t=1}^{\min\{T, d_x\}} \norm{C A^t x_0}.
    \label{eq:stable-truncation}
\end{equation}
\end{lemma}

\begin{proof}
Unrolling the state dynamics gives the exact predictor $y_t = C A^t x_0 + \sum_{i=1}^{t} C A^{i-1} B u_{t-i}$. The truncated predictor uses only the first $h$ terms of the input-driven sum: $\hat y_t^{(h)} = \sum_{i=1}^{h} C A^{i-1} B u_{t-i}$. Their difference splits as
\[
    y_t - \hat y_t^{(h)} \;=\; C A^t x_0 \;+\; \sum_{i=h+1}^{t} C A^{i-1} B u_{t-i}.
\]
For the tail of the input-driven response, the triangle inequality and the stability assumption give
\[
    \Bigl\lVert \sum_{i=h+1}^{t} C A^{i-1} B u_{t-i} \Bigr\rVert \;\leq\; \sum_{i=h+1}^{\infty} \gamma M \rho^{i-1} \beta U \;=\; \frac{M \beta \gamma U}{1 - \rho}\, \rho^h.
\]
Summing the initial-state term over $t = 1, \ldots, T$ and the tail bound over $T$ rounds gives the result.
\end{proof}

Thus, after the initial transient, a stable system can be approximated to error $\varepsilon$ using memory $h = O(\log(T/\varepsilon)/(1 - \rho))$ when $\rho$ is close to one. However, note that the required memory can grow poorly as the stability margin $1 - \rho$ shrinks.

When $\rho(A) = 1$ (marginally stable, e.g., pure sinusoids), this geometric truncation no longer gives a finite-memory approximation with rapidly vanishing tail; classical regret bounds degrade, with the dependence on $d_x$ often becoming linear or worse.

\subsection{Linear Predictors and Online Convex Optimization}
\label{sec:linear-predictors}

The standard convex comparator class for passive prediction is the class of finite-memory linear predictors:
\begin{definition}[Finite-memory linear predictors]
\label{def:linear-predictors}
For memory lengths $h, k$ and norm radius $\kappa$, define $\Pi^L_{h, k, \kappa}$ to be the set of predictors
\begin{equation}
    \hat y_t \;=\; \sum_{i=1}^{h} M_{1, i} \, y_{t-i} + \sum_{j=1}^{k} M_{2, j} \, u_{t-j},
    \qquad
    \sum_{i=1}^{h} \norm{M_{1, i}} + \sum_{j=1}^{k} \norm{M_{2, j}} \leq \kappa.
    \label{eq:linear-predictor}
\end{equation}
\end{definition}

This class is computationally attractive because it is convex in the predictor matrices. Online Gradient Descent (OGD) therefore gives the following learning guarantee. 

\begin{theorem}[OGD for finite-memory predictors; cf.\ {\citealp[Thm.~6.1 and Cor.~9.7]{hazan2025onlinecontrol}}]
\label{thm:ogd-linear-predictors}
Let $K$ be a convex parameter set of diameter $D$, and suppose the loss gradients over $K$ have norm at most $G$. OGD with step sizes $\eta_t = D / (G \sqrt{t})$ satisfies
\begin{equation}
    \sum_{t=1}^{T} \ell_t(M_t) - \min_{M \in K} \sum_{t=1}^{T} \ell_t(M) \;\leq\; \frac{3}{2} G D \sqrt{T}.
    \label{eq:ogd-regret}
\end{equation}
In particular, finite-memory linear predictors are learnable with $O(\sqrt{T})$ regret, with constants depending on the memory length, norm radius, and data bounds.
\end{theorem}

There is also an exact structural reason to study linear predictors. In the noiseless case, every finite-dimensional LDS induces a finite-memory autoregressive predictor.

\begin{theorem}[Cayley--Hamilton representation; cf.\ {\citealp[Thm.~9.4]{hazan2025onlinecontrol}}]
\label{thm:cayley-hamilton}
Consider the noiseless LDS $x_{t+1} = A x_t + B u_t$, $y_t = C x_t$, where $A \in \R^{d_x \times d_x}$ has characteristic polynomial $p_A(\lambda) = \lambda^{d_x} + a_1 \lambda^{d_x - 1} + \cdots + a_{d_x}$. Then there exist matrices $M_{1, i} \in \R^{d_y \times d_y}$ and $M_{2, j} \in \R^{d_y \times d_u}$, $i, j = 1, \ldots, d_x$, such that for every $t > d_x$,
\begin{equation}
    y_t \;=\; \sum_{i=1}^{d_x} M_{1, i} \, y_{t-i} + \sum_{j=1}^{d_x} M_{2, j} \, u_{t-j}.
    \label{eq:cayley-hamilton-predictor}
\end{equation}
Concretely, one may take $M_{1, i} = -a_i I_{d_y}$ and $M_{2, j} = C\bigl(\sum_{i=0}^{j-1} a_i A^{j-1-i}\bigr) B$ (with $a_0 = 1$).
\end{theorem}

\begin{proof}
Write $d = d_x$. Unrolling the state dynamics for $d$ steps gives
\begin{equation}
    x_t \;=\; A^d x_{t-d} + \sum_{j=1}^{d} A^{j-1} B u_{t-j}.
    \label{eq:ch-unroll}
\end{equation}
For each $i = 1, \ldots, d$, similarly $x_{t-i} = A^{d-i} x_{t-d} + \sum_{j=i+1}^{d} A^{j-i-1} B u_{t-j}$, and therefore $A^{d-i} x_{t-d} = x_{t-i} - \sum_{j=i+1}^{d} A^{j-i-1} B u_{t-j}$. By the Cayley--Hamilton theorem, $A^d + a_1 A^{d-1} + \cdots + a_d I = 0$, so $A^d = -\sum_{i=1}^{d} a_i A^{d-i}$. Substituting into~\eqref{eq:ch-unroll} and using the identities for $A^{d-i} x_{t-d}$:
\[
    x_t \;=\; -\sum_{i=1}^{d} a_i x_{t-i} + \sum_{j=1}^{d} \Bigl(\sum_{i=0}^{j-1} a_i A^{j-1-i}\Bigr) B u_{t-j}.
\]
Multiplying both sides by $C$ gives $y_t = -\sum_{i=1}^{d} a_i\, y_{t-i} + \sum_{j=1}^{d} C\bigl(\sum_{i=0}^{j-1} a_i A^{j-1-i}\bigr) B\, u_{t-j}$, which is the claimed recurrence with $M_{1,i} = -a_i I_{d_y}$ and $M_{2,j} = C\bigl(\sum_{i=0}^{j-1} a_i A^{j-1-i}\bigr) B$.
\end{proof}

Theorem~\ref{thm:cayley-hamilton} shows that linear predictors are not just a heuristic, and in the noiseless setting they can exactly represent finite-dimensional LDS, with autoregressive coefficients given precisely by the coefficients of $A$'s characteristic polynomial. The drawback is that the memory length depends on the hidden dimension $d_x$. Lemma~\ref{lem:stable-truncation} removes this dependence for strictly stable systems, but replaces it with dependence on the stability margin $1 - \rho$. Spectral filtering addresses precisely this bottleneck.

\subsection{Spectral Filtering: Compressing Long Memory}
\label{sec:spectral-filtering-prelim}

The Cayley--Hamilton representation (Theorem~\ref{thm:cayley-hamilton}) requires memory $d_x$, the hidden dimension. Spectral filtering, introduced by \citet{hazan2017learning}, compresses this to $O(\log^2 T)$ parameters---independent of both $d_x$ and the stability margin $1 - \rho$---under the following assumptions:
\begin{enumerate}[leftmargin=1.5em, itemsep=0.1em, topsep=0.2em]
    \item $A$ is symmetric (equivalently, diagonalisable with real eigenvalues $\lambda_i \in [-1, 1]$);
    \item $\rho(A) \leq 1$ (marginal stability is allowed);
    \item bounded inputs $\norm{u_t}_2 \leq 1$ and bounded system norms $\norm{B}\norm{C} \leq C_{\mathrm{domain}}$.
\end{enumerate}

The key idea is to replace the raw-history basis $\{u_{t-1}, u_{t-2}, \ldots\}$ with a compressed spectral basis. Because $A$ is symmetric, the impulse response of the LDS decomposes into scalar geometric decays $\lambda_i^t$ with $\lambda_i \in [-1, 1]$. Since $\lambda_i^t$ for negative $\lambda_i$ alternates in sign, one can handle this via a change of basis and reduce to the case $\alpha \in [0, 1]$ (see \citealp[Ch.~11]{hazan2025onlinecontrol} for details). Consider the family of impulse-response vectors
\begin{equation}
    \mu_\alpha \;=\; \begin{bmatrix} 1 & \alpha & \alpha^2 & \cdots & \alpha^{T-1} \end{bmatrix}^\top, \qquad \alpha \in [0, 1],
    \label{eq:impulse-vector}
\end{equation}
and the Hankel-like matrix that integrates over all such responses:
\begin{equation}
    Z_T \;:=\; \int_0^1 \mu_\alpha \mu_\alpha^\top \, d\alpha \in \R^{T \times T},
    \qquad (Z_T)_{ij} = \frac{1}{i + j + 1} \quad (i, j = 0, \ldots, T - 1).
    \label{eq:hankel-matrix}
\end{equation}
The eigenvalues of $Z_T$ decay exponentially \citep[Thm.~11.1]{hazan2025onlinecontrol}:
\begin{equation}
    \sigma_k(Z_T) \;\leq\; 4 \pi^2 \exp\!\left(-\frac{k}{\log T}\right) \qquad (k, T > 1).
    \label{eq:hankel-decay}
\end{equation}
This rapid decay is the reason spectral filtering is efficient: the top $h = O(\log^2 T / \varepsilon)$ eigenvectors $\phi_1, \ldots, \phi_h$ of $Z_T$ suffice to $\varepsilon$-approximate every impulse response in the family $\{\mu_\alpha : \alpha \in [0, 1]\}$, and therefore every scalar component of the LDS's response. The spectral predictor projects the input history onto this basis:
\begin{equation}
    \hat y_t \;=\; \sum_{i=1}^{h} M_i \, \langle \tilde u_t, \phi_i \rangle,
    \label{eq:spectral-predictor}
\end{equation}
where $\tilde u_t = (u_{t-1}, u_{t-2}, \ldots, u_1, 0, \ldots, 0)$. The coefficients $M_i$ are learned online via OGD (Theorem~\ref{thm:ogd-linear-predictors}), yielding sublinear regret with $h = O(\log^2 T)$ parameters.

% \paragraph{Limitations: asymmetry and the unit circle.}
% The preceding development requires $A$ to be symmetric so that the eigenvalues are real and the impulse responses have the scalar form~\eqref{eq:impulse-vector}. When $A$ is asymmetric, eigenvalues may be complex, and the impulse responses involve oscillating terms $\abs{\lambda}^t e^{i \arg(\lambda) t}$ that cannot be captured by the real-valued basis $\{\mu_\alpha\}$. \citet[Thm.~12.1]{hazan2025onlinecontrol} gives a cyclic-permutation LDS construction showing that any online learner incurs expected cumulative squared loss $\Omega(d_x)$ over $T \geq d_x$ rounds for general asymmetric systems---a fundamental barrier. The universal-sequence-preconditioning technique we describe in Section~\ref{sec:precondition} addresses this barrier: by polynomially compressing the spectral radius before learning, USP extends sublinear regret guarantees to asymmetric systems whose eigenvalue arguments are at most $O(1/\log^2 T)$.

\section{Universal Sequence Preconditioning}
\label{sec:precondition}
\label{sec:chebyshev}
%==============================================================================

\citet{marsden2024universal} introduced a polynomial-preconditioning framework that yields the first sublinear, hidden-dimension-independent regret bounds for online prediction in marginally-stable LDSs. Convolving the observation sequence with the coefficients of a fixed degree-$d$ monic polynomial $p$ is algebraically equivalent to multiplying the hidden dynamics by $p(A)$; choosing $p$ to be the monic Chebyshev polynomial then minimax-optimally compresses the spectral radius the downstream learner faces. 

\subsection{Polynomial Preconditioning of LDS Observations}
\label{sec:lds-identity}

Consider an LDS as in~\eqref{eq:lds} with noise terms suppressed for exposition, so that $y_t = C A^t x_0$. Write a monic degree-$d$ polynomial in descending powers as $p(z) = \sum_{i=0}^{d} c_i z^{d-i}$ with $c_0 = 1$. The preconditioned observation is
\begin{equation}
    \tilde y_t \;:=\; \sum_{i=0}^{d} c_i\, y_{t-i} \;=\; C \cdot p(A) \cdot A^{t-d} x_0,
    \label{eq:mh-identity}
\end{equation}
where the second equality follows by substituting $y_{t-i} = C A^{t-i} x_0$ and factoring out $A^{t-d}$. The identity replaces the implicit dependence of $\tilde y_t$ on $A^{t-d}$ by $p(A) \cdot A^{t-d}$. If $A$'s real spectrum lies in $[-\rho, \rho]$, the relevant bound is
\begin{equation}
    \rho\big( p(A) \big) \;\leq\; \sup_{\lambda \in [-\rho,\rho]} \abs{p(\lambda)}.
    \label{eq:rho-bound}
\end{equation}
Minimising the right-hand side over monic $p$ of degree $d$ is the classical Chebyshev minimax problem (Theorem~\ref{thm:chebyshev-minimax}), and its solution $p(\lambda) = \rho^d \tilde T_d(\lambda/\rho)$ achieves $\sup_{[-\rho,\rho]} \abs{p} \leq \rho^d \cdot 2^{1-d}$. Preconditioning by Chebyshev coefficients therefore compresses the effective spectral radius seen by a downstream predictor by a factor of $2^{1-d}$ in the worst case over $A \in \{A : \mathrm{spec}(A) \subset [-\rho,\rho]\}$.



\subsection{Chebyshev Polynomials}
\label{sec:chebyshev-prelim}

The \emph{Chebyshev polynomials of the first kind}, denoted $T_d(z)$ for $d \in \Z_{\geq 0}$, are defined by the recurrence
\begin{equation}
    T_0(z) = 1, \quad T_1(z) = z, \quad T_{d+1}(z) = 2 z \, T_d(z) - T_{d-1}(z),
    \label{eq:cheb-recurrence}
\end{equation}
or equivalently by the trigonometric identity $T_d(\cos \theta) = \cos(d \theta)$. The first several polynomials are $T_2(z) = 2z^2 - 1$, $T_3(z) = 4z^3 - 3z$, $T_4(z) = 8z^4 - 8z^2 + 1$. The leading coefficient of $T_d$ is $2^{d-1}$, so the \emph{monic Chebyshev polynomial} is $\tilde T_d(z) = T_d(z) / 2^{d-1}$. For $d = 4$,
\[
    \tilde T_4(z) = z^4 - z^2 + \tfrac{1}{8}.
\]
Two further facts will be used below. First, the trigonometric definition makes Chebyshev polynomials \emph{equioscillate} on $[-1,1]$: $\tilde T_d$ attains its $L^\infty$ norm $2^{1-d}$ with alternating signs at the $d{+}1$ extremal points $x_k = \cos(k\pi/d)$, $k = 0, \ldots, d$. Second, the monomial-basis coefficients of $\tilde T_d$ have $\ell^1$ norm $\norm{\boldsymbol{a}^{(d)}}_1 \leq 2^{0.3d}$ for all $d \geq 1$ \citep[Lem.~3.2]{marsden2024universal}; this controls the noise gain of the preconditioning filter and enters the regret bound of Section~\ref{sec:usp-regret} as the comparator-class diameter.

\subsection{Chebyshev Minimax}
\label{sec:cheb-minimax}

\begin{theorem}[Chebyshev minimax; cf.\ \citealp{davis1975interpolation}]
\label{thm:chebyshev-minimax}
Among all real monic polynomials of degree $d \geq 1$, the monic Chebyshev polynomial $\tilde T_d$ uniquely minimises $\sup_{x \in [-1,1]} \abs{q(x)}$:
\[
    \tilde T_d \;=\; \argmin_{q \text{ monic, } \deg q = d} \; \max_{x \in [-1,1]} \abs{q(x)}, \qquad \max_{x \in [-1,1]} \abs{\tilde T_d(x)} = 2^{1-d}.
\]
\end{theorem}

\begin{proof}[Proof sketch]
By the trigonometric definition $T_d(\cos\theta) = \cos(d\theta)$, $T_d$ ranges over $[-1,1]$ on $[-1,1]$ with $d+1$ alternating extrema $\tilde T_d(x_k) = (-1)^k \cdot 2^{1-d}$ at $x_k = \cos(k\pi/d)$, $k = 0, \ldots, d$. Suppose for contradiction that some monic $q$ of degree $d$ has $\sup_{[-1,1]} \abs{q} < 2^{1-d}$. Then $r := \tilde T_d - q$, of degree $\leq d-1$, takes the same sign as $\tilde T_d$ at each $x_k$ and so alternates in sign across $d+1$ points; the intermediate-value theorem gives at least $d$ zeros, forcing $r \equiv 0$, a contradiction.
\end{proof}

\begin{corollary}[Worst-case spectral compression]
\label{cor:spectral-compression}
For diagonalisable $A$ with real spectrum in $[-\rho, \rho]$, the choice $q(\lambda) = \rho^d \tilde T_d(\lambda/\rho)$ minimises $\sup_A \rho(q(A))$ over monic polynomials of degree $d$, attaining
\[
    \sup \rho\big(q(A)\big) \;=\; \sup_{\lambda \in [-\rho, \rho]} \abs{q(\lambda)} \;\leq\; \rho^d \cdot 2^{1-d}.
\]
\end{corollary}

\subsection{The USP Algorithm}
\label{sec:usp-algorithm}

Leveraging the Chebyshev minimax property, we can state the USP algorithm of \citet{marsden2024universal}. The starting point is the three-term decomposition of the LDS output. For a system~\eqref{eq:lds} with inputs $u_1, \ldots, u_T$ and monic preconditioning polynomial $p_n^c(z) = \sum_{i=0}^{n} c_i z^{n-i}$ with $c_0 = 1$, one obtains
\begin{equation}
    y_t \;=\; \underbrace{-\sum_{i=1}^{n} c_i\, y_{t-i}}_{\aleph_0\text{ (preconditioning)}}
    \;+\; \underbrace{\sum_{s=0}^{n-1} \sum_{i=0}^{s} c_i\, C A^{s-i} B\, u_{t-s}}_{\aleph_1\text{ (learnable)}}
    \;+\; \underbrace{\sum_{s=0}^{t-n-1} C\, p_n^c(A)\, A^s B\, u_{t-n-s}}_{\aleph_2\text{ (residual error)}}.
    \label{eq:three-term}
\end{equation}
The term $\aleph_0$ is the universal preconditioning component: it depends only on the coefficients $c_{1:n}$ (not on $A$, $B$, or $C$) and can be computed from observed outputs alone. The term $\aleph_1$ involves a bounded number of parameters that can be learned via convex regression. The critical term $\aleph_2$ contains $p_n^c(A)$; if $p_n^c$ is chosen to have small supremum on the spectrum of $A$, this error term is small.

Algorithm~\ref{alg:usp-general} states the general USP framework. Any sequence prediction model $f_\theta$ can serve as the base learner: at each round, USP subtracts the preconditioning term $\aleph_0 = -\sum c_j y_{t-j}$ from the target and delegates the remainder to $f_\theta$, which is updated online. The preconditioning coefficients $c_{1:n}$ are fixed (not learned) and independent of the underlying system.

\begin{algorithm}[t]
\caption{Universal Sequence Preconditioning \citep[Alg.~1]{marsden2024universal}}
\label{alg:usp-general}
\begin{algorithmic}[1]
\REQUIRE Sequence prediction model $f_\theta$ with initial parameters $\theta_0$; loss function $\ell(\cdot)$; preconditioning coefficients $c_{1:n}$.
\FOR{$t = 1$ \TO $T$}
    \STATE Receive input $u_t$.
    \STATE Predict $\hat y_t = -\sum_{j=1}^{n} c_j\, y_{t-j} + f_{\theta_t}(u_{1:t},\, y_{1:t-1})$.
    \STATE Observe true output $y_t$; suffer loss $\ell_t(\hat y_t, y_t)$.
    \STATE Update: $\theta_{t+1} \leftarrow \mathrm{proj}_{\mathcal{K}}\!\bigl(\theta_t - \eta_t \nabla_\theta \ell_t(\hat y_t, y_t)\bigr)$.
\ENDFOR
\end{algorithmic}
\end{algorithm}

Algorithm~\ref{alg:usp-regression} instantiates this framework with convex regression as the base learner. The learnable parameters are matrices $Q_0, \ldots, Q_{n-1}$ that approximate the $\aleph_1$ term; the residual $\aleph_2$ is treated as an unlearnable error. This specific instantiation is the one for which Marsden and Hazan derive formal regret bounds (Section~\ref{sec:usp-regret}).

\begin{algorithm}[t]
\caption{USP for Regression \citep[Alg.~2]{marsden2024universal}}
\label{alg:usp-regression}
\begin{algorithmic}[1]
\REQUIRE Initial parameters $Q_0$; preconditioning coefficients $c_{0:n}$ from the $n$-th degree monic Chebyshev polynomial; convex constraint set $\mathcal{K} = \{(Q_0, \ldots, Q_{n-1}) : \norm{Q_j} \leq C_{\mathrm{domain}} n \norm{\boldsymbol{c}}_1\}$.
\STATE Assert $c_0 = 1$.
\FOR{$t = 1$ \TO $T$}
    \STATE Receive input $u_t$.
    \STATE Predict $\hat y_t(Q^t) = -\sum_{i=1}^{n} c_i\, y_{t-i} + \sum_{j=0}^{n-1} Q_j^t\, u_{t-j}$.
    \STATE Observe $y_t$; suffer loss $\ell_t(Q^t) = \norm{\hat y_t(Q^t) - y_t}_1$.
    \STATE Update: $Q^{t+1} \leftarrow \mathrm{proj}_{\mathcal{K}}\!\bigl(Q^t - \eta_t \nabla_Q \ell_t(Q^t)\bigr)$.
\ENDFOR
\end{algorithmic}
\end{algorithm}

\citet{marsden2024universal} also introduce a more powerful instantiation that combines preconditioning with spectral filtering (their Algorithm~3). Spectral filtering learns both $\aleph_1$ and $\aleph_2$, yielding a better regret rate of $\tilde O(T^{-3/13})$ compared to $\tilde O(T^{-2/13})$ for regression alone (Theorems~\ref{thm:usp-general} and~\ref{thm:usp-chebyshev}). In both cases, the preconditioning polynomial shrinks the learnable domain, making the online optimisation problem easier.

\subsection{The USP Regret Bound}
\label{sec:usp-regret}

The central theoretical result of \citet{marsden2024universal} is that USP with Chebyshev coefficients achieves sublinear, hidden-dimension-independent average loss for online prediction in marginally-stable LDSs. We state the general cumulative-loss bound first (note: this bounds total prediction loss, not regret relative to a comparator), then the Chebyshev specialisation.

\begin{theorem}[USP cumulative loss; general polynomial {\citep[Thm.~D.1]{marsden2024universal}}]
\label{thm:usp-general}
Let $\{u_t\}_{t=1}^T \subset \R^{d_{\mathrm{in}}}$ with $\norm{u_t}_2 \leq 1$, and let $\{y_t\}_{t=1}^T$ be the output of an LDS $(A, B, C)$ with $\norm{B}\norm{C}\kappa \leq C_{\mathrm{domain}}$, where $\kappa = \norm{P}\norm{P^{-1}}$ for a diagonalisation $A = P D P^{-1}$. Then Algorithm~\ref{alg:usp-regression} with any monic degree-$n$ polynomial $p_n^c$ satisfies
\begin{equation}
    \sum_{t=1}^{T} \norm{\hat y_t - y_t}_1 \;\leq\; C_{\mathrm{domain}} \!\left[\; \tfrac{3}{2}\, n^2 \sqrt{d_{\mathrm{out}}}\, \norm{\boldsymbol{c}}_1 \sqrt{T} \;+\; \max_{\lambda \in \mathrm{spec}(A)} \abs{p_n^c(\lambda)}\, T^2 \;\right].
    \label{eq:usp-regret-general}
\end{equation}
\end{theorem}

\begin{proof}[Proof outline]
The bound is the sum of two terms. The first is the online-gradient-descent (OGD) regret $\frac{3}{2} G D \sqrt{T}$ of \citet[Thm.~6.1]{hazan2025onlinecontrol} applied to the loss sequence $\ell_t(Q) = \norm{\hat y_t(Q) - y_t}_1$ over the convex set $\mathcal{K}$. The diameter is $D \leq 2 C_{\mathrm{domain}} n \norm{\boldsymbol{c}}_1$ (each $\norm{Q_j} \leq C_{\mathrm{domain}} n \norm{\boldsymbol{c}}_1$ by the constraint set) and the gradient bound is $G \leq n \sqrt{d_{\mathrm{out}}}$ (since $\norm{\nabla_{Q_i} \ell_t}_F \leq \sqrt{d_{\mathrm{out}}} \norm{u_{t-i}}_2 \leq \sqrt{d_{\mathrm{out}}}$). The second term bounds the comparator's loss. Setting $\hat Q_s = \sum_{i=0}^{s} c_i C A^{s-i} B$ makes $\hat Q \in \mathcal{K}$ (since $C_{\mathrm{domain}} \geq \norm{C}\norm{B}\kappa$), and the prediction error at $\hat Q$ reduces to the $\aleph_2$ tail: $\norm{\hat y_t(\hat Q) - y_t}_1 \leq \norm{C}\norm{B}\kappa \cdot \max_\lambda \abs{p_n^c(\lambda)} \cdot T$. Summing over $T$ rounds gives the $T^2$ term.
\end{proof}

% The two terms in~\eqref{eq:usp-regret-general} exhibits a tradeoff: $\norm{\boldsymbol{c}}_1$ (which controls the OGD regret) competes with $\max_\lambda \abs{p_n^c(\lambda)}$ (which controls the comparator error). 

\begin{theorem}[USP regression with Chebyshev coefficients {\citep[Thm.~2.1]{marsden2024universal}}]
\label{thm:usp-chebyshev}
Under the hypotheses of Theorem~\ref{thm:usp-general}, if additionally $\max_{j \in [d_x]} \abs{\arg(\lambda_j)} \leq (64 \log_2(8T^{3/2}/(3\sqrt{d_{\mathrm{out}}})))^{-2}$ and the coefficients $c_{0:n}$ are those of the $n$-th monic Chebyshev polynomial, then
\[
    \frac{1}{T} \sum_{t=1}^{T} \norm{\hat y_t - y_t}_1 \;\leq\; \frac{9\, C_{\mathrm{domain}} \sqrt{d_{\mathrm{out}}}\, \log_2^2(3T)}{T^{2/13}}.
\]
\end{theorem}

\begin{proof}[Proof sketch]
By Theorem~\ref{thm:usp-general}, it suffices to bound $\norm{\boldsymbol{c}}_1$ and $\max_\lambda \abs{p_n^c(\lambda)}$ for the monic Chebyshev polynomial. Lemma~3.2 of \citet{marsden2024universal} gives $\norm{\boldsymbol{c}}_\infty \leq 2^{0.3n}$, hence $\norm{\boldsymbol{c}}_1 \leq (n+1) \cdot 2^{0.3n}$. For the spectral bound, Lemma~3.1 of \citet{marsden2024universal} shows that $\max_\lambda \abs{\tilde T_n(\lambda)} \leq 2^{-(n-2)}$ when each eigenvalue $\lambda$ satisfies $\abs{\arg(\lambda)} \leq 1/(64n^2)$. Substituting into~\eqref{eq:usp-regret-general} and balancing the two terms by setting $n = \tfrac{15}{13} \log_2 T$ yields the $T^{-2/13}$ average-loss rate.
\end{proof}

\noindent For comparison, vanilla regression without preconditioning---i.e., setting $n = 0$ in Theorem~\ref{thm:usp-general} so that $\max_\lambda \abs{p_0^c(\lambda)} = 1$---achieves cumulative regret $O(C_{\mathrm{domain}} \sqrt{d_{\mathrm{out}}}\, T^{5/2})$ \citep[p.~5]{marsden2024universal}, which is \emph{not} sublinear in $T$. USP with spectral filtering (Algorithm~3 of \citet{marsden2024universal}) improves the rate further to $\tilde O(T^{-3/13})$ by learning the $\aleph_2$ term as well, at the cost of a more complex spectral basis. Both rates are \emph{independent of the hidden dimension} $d_x$---the signature contribution of USP. These bounds apply to the online LDS setting of \citet{marsden2024universal}; our empirical method (Section~\ref{sec:method}) uses the same Chebyshev filter but in a different setting (pretrained transformer, batch training), so the formal guarantees do not directly transfer.

\section{Patch-based Transformers for Forecasting}
\label{sec:patch-transformers}
%==============================================================================

\emph{Time series foundation models} are large pretrained models that produce zero-shot forecasts for arbitrary unseen series without task-specific fine-tuning. Like language foundation models, they are trained on a large and diverse corpus and transfer the learned temporal patterns to new series at inference time. The dominant architecture is the \emph{patch-based transformer}: the input series is segmented into fixed-length patches, each patch is projected into an embedding vector, and a stack of transformer layers models inter-patch dependencies via self-attention. The output head decodes these representations into a predictive distribution over future time steps. Several recent models follow this paradigm, including PatchTST~\citep{nie2023patchtst}, Moirai~\citep{woo2024unified}, TimesFM~\citep{das2024decoder}, and Chronos~\citep{ansari2024chronos}, differing primarily in training objective, patch encoding, and output parameterisation.

Our experiments are conducted on \emph{Moirai-2 Small}~\citep{liu2024moirai2}, the open-source\footnote{Available on the HuggingFace Hub as \texttt{Salesforce/moirai-2.0-R-small}; the architecture and training recipe follow the description in \citet{liu2024moirai2}.} dense, decoder-only successor to the original Moirai encoder~\citep{woo2024unified}. We chose Moirai-2 because it is a small ($11.4$M parameter), publicly reproducible, state-of-the-art universal forecaster. It is small enough to train from scratch on a single H100 within wall-clock budgets compatible with five-seed ablations, yet competitive with much larger models on the GIFT-Eval benchmark. Our preconditioning method modifies only the input projection of this model.

\subsection{Patches and the Per-Patch Projection}
\label{sec:patches}

A patch-based transformer forecaster~\citep{nie2023patchtst, woo2024unified, liu2024moirai2} proceeds as follows. Given a univariate history $x_{1:T}$, partition it into non-overlapping patches $\mathbf{p}_i = (x_{(i-1)P+1}, \ldots, x_{iP})$ of length $P$ for $i=1, \ldots, T/P$. A per-patch projection (in Moirai-2, a residual block of two linear layers with a SiLU nonlinearity) maps each patch to an embedding $\mathbf{e}_i \in \R^{d_{\text{model}}}$. These embeddings are fed to a transformer block stack whose output is decoded into a predictive distribution over future patches via a per-patch projection head. Two features of this architecture are salient for our development:
\begin{itemize}[leftmargin=1.5em, itemsep=0.1em, topsep=0.2em]
    \item \emph{Cross-patch information must be learned via attention.} Before the first self-attention operation, $\mathbf{e}_i$ is a function of $\mathbf{p}_i$ alone; it contains no information about $\mathbf{p}_{i-1}$, $\mathbf{p}_{i+1}$, or more distant patches. Patterns whose natural temporal scale exceeds $P$ time steps (e.g., daily periodicities in $15$-minute data, spanning $96$ steps or $6$ patches at $P=16$) must be discovered end-to-end by the attention layers above the projection.
    \item \emph{The per-patch projection is narrow.} For Moirai-2 Small ($P=16$, $d_{\text{model}}=384$), the projection's input dimension is $2P = 32$ (target plus observation mask), so it has roughly $32 \cdot 384 \approx 12{,}000$ parameters out of $11.4$M total ($\sim 0.1\%$). Adding a third $P$-dimensional input channel costs $\approx 12$K additional parameters ($P \cdot d_{\text{model}} = 6{,}144$ in the first linear layer plus $\approx 6{,}000$ in the residual path); altering downstream layers, in contrast, is much more expensive.
\end{itemize}
These two observations motivate \emph{input-side} preconditioning: a fixed preconditioning filter applied before patching can inject cross-patch information into each $\mathbf{e}_i$ at negligible parameter cost, leaving the rest of the network untouched.

\subsection{The Moirai-2 Small Architecture}
\label{sec:moirai-arch}

Moirai-2 Small is a decoder-only transformer with $L = 6$ layers, hidden dimension $d_{\text{model}} = 384$, feed-forward dimension $d_{\text{ff}} = 1024$, $h = 6$ attention heads, and patch size $P = 16$. Each transformer block applies pre-norm self-attention with rotary position embeddings (RoPE)~\citep{su2024rope} and a SwiGLU feed-forward network~\citep{shazeer2020glu}, with RMSNorm~\citep{zhang2019rms} replacing LayerNorm throughout. A binary attention bias~\citep{liu2024moirai2} discriminates between positions in the observed prefix and positions in the prediction horizon. Inputs are $z$-scored at the sequence level before patching using a packed-sequence variant of standard $z$-score scaling. Three features of the architecture are particularly relevant to the rest of this paper:
\begin{itemize}[leftmargin=1.5em, itemsep=0.1em, topsep=0.2em]
    \item \emph{Decoder-only with causal mask.} Unlike the masked-language-model encoder used in Moirai 1.0~\citep{woo2024unified}, Moirai-2 attends to past patches only. Predictions for patch $i$ depend on $\mathbf{e}_1, \ldots, \mathbf{e}_i$ but not on later patches.
    \item \emph{Multi-token prediction head.} The output projection emits, for each input patch, predictions for the next $K = 4$ patches at $9$ quantile levels $\{0.1, 0.2, \ldots, 0.9\}$, yielding $K \cdot 9 \cdot P = 576$ output values per patch.
    \item \emph{Per-patch input residual block.} The input projection is a two-layer residual MLP $\mathrm{ResBlock}_{2P \to d_{\text{model}}}$ acting on the concatenation $[\mathbf{p}_i; \mathbf{m}_i]$ of patch and observation mask. This is the only layer that touches raw patch values; it is also the only layer modified by our method (Section~\ref{sec:method} below).
\end{itemize}

\subsection{Pretraining Objective and Optimization}
\label{sec:pretraining-objective}

Each training sample is a packed sequence of patched univariate series. A random \emph{prefix ratio} (default $0.3$) determines how much of the sequence is treated as observed context; the remainder is the prediction window. Within the prediction window, the target patches are replaced by a learnable mask token, and the model is trained to predict their values. Additionally, with probability $\texttt{mask\_ratio} = 0.5$, individual patches inside the prediction window are corrupted to encourage robustness to missing observations.

The training loss is the \emph{packed quantile MAE loss},
\begin{equation}
    \mathcal{L} \;=\; \frac{1}{N \cdot |\mathcal{Q}|} \sum_{n,t,q} \rho_q\!\left(\, y_{n,t} - \hat y_{n,t}^{(q)} \,\right),
    \label{eq:quantile-loss}
\end{equation}
where the sum runs over patches in the prediction window of all packed samples, $\mathcal{Q} = \{0.1, \ldots, 0.9\}$ is the set of quantile levels, $\hat y_{n,t}^{(q)}$ is the model's $q$-quantile prediction, and $\rho_q(u) = u(q - \mathbf{1}[u < 0])$ is the standard pinball loss~\citep{koenker1978regression}. We use AdamW~\citep{loshchilov2019adamw} with $\beta_1 = 0.9$, $\beta_2 = 0.98$, $\epsilon = 10^{-6}$, weight decay $0.1$, peak learning rate $\eta = 10^{-3}$, $1{,}000$-step linear warmup, and cosine annealing to zero. Training is performed in bf16 mixed precision with batch size $256$ on a single H100 GPU. Anomaly filtering~\citep{liu2024moirai2} resamples training windows whose normalised values exceed a $z$-score threshold of $8.0$ to suppress extreme outliers in LOTSA. The full set of hyperparameters appears in Appendix~\ref{app:methodology}.

\subsection{Training Corpus: LOTSA}
\label{sec:lotsa}

Moirai-2 is pretrained on the \emph{Large-scale Open Time Series Archive} (LOTSA), introduced by \citet{woo2024unified}, which aggregates $\sim 27$\,billion observations across $9$ broad domains: energy, transport, climate, cloud operations, web, sales, healthcare, nature, and economic/financial. Domains are unbalanced, with a few datasets (e.g., Buildings-900K, ERA5) dominating raw observation counts. Therefore, Moirai-2 uses \emph{variate-proportional weighted sampling}, where for each dataset, the sampling weight is inversely proportional to its raw observation count and proportional to the number of distinct variates, which roughly equalises the contribution of large multivariate datasets and small univariate ones. We use the released LOTSA v1 weighting scheme without modification (the \texttt{lotsa\_v1\_moirai2} configuration in the official codebase). Throughout training, each sample is normalised by its own mean and standard deviation (computed within the observed prefix) before patching.

\subsection{Inference and Forecasting}
\label{sec:inference}

At inference, the model is presented with the full observed history followed by mask tokens for the prediction window. Because the multi-token head emits a $K{=}4$-patch forecast for every input patch, predictions for the prediction-window patches are read directly from the corresponding output positions; no autoregressive rollout is required for horizons of up to $K \cdot P = 64$ time steps. For longer horizons, the predicted patches are appended to the context and the procedure repeats. Each forecast at horizon $h$ produces a vector of $9$ quantile estimates at each time step; for our headline metric (MASE), we use the median ($q = 0.5$) as the point forecast.

\subsection{Evaluation: GIFT-Eval}
\label{sec:gifteval}

We evaluate on \textsc{GIFT-Eval}~\citep{vijay2024gifteval}, an open benchmark of $97$ dataset $\times$ horizon configurations spanning $24$ datasets and $7$ frequency classes (sub-hourly through yearly), curated to test the zero-shot generalization claim of universal forecasters. Each configuration specifies a held-out split, prediction horizon, and patch-size convention; we follow the recommended convention of $P = 8$ for quarterly data and $P = 32$ for all other frequencies, with context length $4{,}000$. Note that these evaluation-time patch sizes differ from the architecture's default $P = 16$ (Section~\ref{sec:moirai-arch}); Moirai-2 supports variable patch sizes at inference. Our preconditioning filter stride $s$ is always set equal to the evaluation-time patch size $P$. The remainder of this subsection rigorously defines the metric we report.

\paragraph{Mean Absolute Scaled Error (MASE).}
For a single test series with point forecast $\hat y_{t}$ and ground-truth $y_{t}$ over a prediction window of length $H$, the \emph{Mean Absolute Scaled Error} \citep{hyndman2006another} is defined as
\begin{equation}
    \mathrm{MASE}(\hat y, y) \;=\; \frac{\frac{1}{H}\sum_{t=1}^{H} \abs{y_t - \hat y_t}}{\frac{1}{N - m}\sum_{t = m+1}^{N} \abs{y^{\text{train}}_t - y^{\text{train}}_{t-m}}},
    \label{eq:mase-def}
\end{equation}
where $y^{\text{train}}_{1:N}$ is the in-sample (training) portion of the same series and $m$ is the seasonal period (e.g., $m = 24$ for hourly data with daily seasonality, $m = 1$ for non-seasonal series). The denominator is the in-sample mean absolute error of the \emph{seasonal naive} predictor $\hat y_t^{\text{snaive}} = y_{t-m}$, providing a series-specific scaling that makes the metric unitless and comparable across series of different magnitudes. A MASE below $1$ means the model beats the seasonal-naive predictor on that series in mean absolute error; above $1$ means it does not.

\paragraph{Aggregation across a configuration.}
A GIFT-Eval ``configuration'' $c$ is a (dataset, prediction-horizon) pair. The benchmark evaluates the model on a fixed test split of multiple windows from each dataset; let $\mathcal{W}_c$ denote this set of test windows. The configuration-level raw MASE is the mean across windows:
\begin{equation}
    \mathrm{MASE}(\text{model}, c) \;=\; \frac{1}{\abs{\mathcal{W}_c}} \sum_{(\hat y, y) \in \mathcal{W}_c} \mathrm{MASE}(\hat y, y).
\end{equation}

\paragraph{Normalised MASE.}
Because absolute MASE values vary by orders of magnitude across configurations (some series are easy to forecast, some are not), the GIFT-Eval leaderboard divides each configuration's MASE by that configuration's seasonal-naive MASE before aggregating across configurations:
\begin{equation}
    \mathrm{MASE}^*(\text{model}, c) \;=\; \frac{\mathrm{MASE}(\text{model}, c)}{\mathrm{MASE}(\text{seasonal naive}, c)}.
    \label{eq:normalised-mase}
\end{equation}
A configuration-level $\mathrm{MASE}^*$ below $1$ means the model beats seasonal-naive on that configuration; above $1$ means it does not. Note that $\mathrm{MASE}^*$ is the ratio of the model's MASE to the seasonal-naive MASE, both computed using \eqref{eq:mase-def}; the seasonal-naive denominator therefore cancels, and $\mathrm{MASE}^*(\text{model}, c)$ is equivalently the ratio of the model's window-mean absolute error to the seasonal-naive predictor's window-mean absolute error on configuration $c$.

\paragraph{Geometric-mean aggregation.}
The headline aggregate is the geometric mean of $\mathrm{MASE}^*$ across all $C = 97$ configurations:
\begin{equation}
    \overline{\mathrm{MASE}^*}(\text{model}) \;=\; \left(\prod_{c=1}^{C} \mathrm{MASE}^*(\text{model}, c)\right)^{1/C} \;=\; \exp\!\left(\frac{1}{C}\sum_{c=1}^{C} \log \mathrm{MASE}^*(\text{model}, c)\right).
    \label{eq:geomean}
\end{equation}
The geometric mean is preferred over the arithmetic mean because it is symmetric in the log-domain ($\log\overline{\mathrm{MASE}^*}$ is the arithmetic mean of $\log\mathrm{MASE}^*$), so the aggregate cannot be dominated by a single configuration on which one model has an extreme MASE. Equivalently, it averages \emph{relative} performance across configurations. 

% As a reference point, our retrained Moirai-2 Small \Baseline{} attains $\overline{\mathrm{MASE}^*} \approx 0.862$ at $10$K steps; the official Moirai-2.0-R-Small release attains $\approx 0.728$ on the same metric (Appendix~\ref{app:official-moirai}).

\paragraph{Reporting convention.}
We report cross-seed mean and standard deviation of $\overline{\mathrm{MASE}^*}$ together with per-configuration paired-comparison win counts (out of $97$ per seed; $485$ pooled across the $5$ seeds we use). Headline relative changes between methods are reported as $\Delta\% = \big(\overline{\mathrm{MASE}^*}(\text{treatment}) - \overline{\mathrm{MASE}^*}(\text{control})\big) / \overline{\mathrm{MASE}^*}(\text{control}) \cdot 100$, so a negative $\Delta\%$ means the treatment achieves lower MASE (an improvement). 


% We do not report $p$-values for the per-configuration win counts: although a paired sign test would assign extremely small values to our pooled wins, the $5 \times 97 = 485$ paired observations are not independent (each of the $97$ configurations is reused across all $5$ seeds with the same training data and evaluation setup), so the iid-binomial null is mis-specified and any nominal $p$-value should be read with caution.

\subsection{The Polynomial Preconditioning Channel}
\label{sec:method}
In addition to the formal guarantees stated in prior sections, \citet{marsden2024universal} found empirically that USP with fixed Chebyshev or Legendre coefficients reduces prediction error on synthetic linear, nonlinear, and deep-RNN-generated sequences and on the ETTh1 benchmark. A natural question is whether these techniques transfer to patch-based time series forecasters.

One natural approach might be to precondition the target, train the predictor to emit $\hat{\tilde y}_{t+1}$, and invert at inference to recover $\hat y_{t+1}$. However, this fails at multi-step horizons because the inversion is a causal recurrence
\[
\hat y_t \;=\; \hat{\tilde y}_t \;-\; \sum_{k=1}^{d} c_k\, \hat y_{t-k},
\]
in which each recovered value $\hat y_t$ depends on previously recovered values $\hat y_{t-1}, \ldots, \hat y_{t-d}$. Any error in $\hat y_{t-k}$ feeds forward into all subsequent inversions, so prediction errors compound geometrically across the rollout rather than staying local to each step.

Instead, we leave the forecast target unchanged and compute the preconditioning residual $r_t$ (Eq.~\ref{eq:residual-channel} below) as an auxiliary input channel concatenated alongside the raw target and the observation mask. Because the model never operates in preconditioned space at inference, no inversion is needed and error accumulation is avoided.

\paragraph{Specialization to patch-based forecasting.}
Our setting differs from the LDS online-prediction setting in three ways: (i) the forecaster is a pretrained transformer, not an online learner; (ii) the data-generating process is unknown and not assumed to be a finite-dimensional LDS; (iii) rather than \emph{replacing} the target sequence by its preconditioned version (which would require inverting the convolution at inference and is incompatible with multi-step rollout), we \emph{concatenate} the preconditioned sequence as an auxiliary input channel. Concretely, given the coefficients $c_{0:d}$ of the monic Chebyshev polynomial (with $c_0 = 1$), we apply the causal filter $\sum_{i=0}^{d} c_i x_{t-is}$ at stride $s$ to the input series $x$ and feed the model the residual
\begin{equation}
    r_t \;=\; \sum_{k=1}^{d} c_k\, x_{t-ks},
    \label{eq:residual-channel}
\end{equation}
alongside the original $x$. We work with the residual rather than the full filter output because the original signal is already provided by the unmodified target channel; the residual captures the \emph{novel} cross-patch information injected by preconditioning. Despite the differences from the original LDS setting, the spectral intuition transfers: a Chebyshev preconditioner shapes the input's spectrum minimax-optimally, and a patch-based transformer facing a signal whose spectrum has been shaped to highlight multi-patch correlations is plausibly better-equipped to capture them.

We now describe the architectural modification that delivers the preconditioning channel of Eq.~\eqref{eq:residual-channel} into the model. Let $x \in \R^T$ be a univariate input series, normalised to zero mean and unit variance over the observed prefix. We form the residual $r_t = \sum_{k=1}^{d} c_k\, x_{t-ks}$, where $c_{1:d}$ are the non-leading coefficients of the monic Chebyshev polynomial $\tilde T_d$. Because the filter uses only past lags ($k \geq 1$), $r_t$ depends only on $x_{t-s}, x_{t-2s}, \ldots, x_{t-ds}$ and never on future values. During training, positions in the prediction window where the target is masked are zeroed before filtering, so no future information leaks into $r_t$. During long-horizon autoregressive inference, when the model must roll out beyond the $K$-patch prediction window, the previously predicted values are used in place of ground truth when computing $r_t$ for the extended context; channel dropout during training regularises against the resulting noise (see below). The residual is then partitioned into patches $\mathbf{r}_i$ of length $P$ aligned with the target patches $\mathbf{p}_i$. The per-patch input projection is widened to accept the augmented triple $(\mathbf{p}_i, \mathbf{m}_i, \mathbf{r}_i)$:
\begin{equation}
    \mathbf{e}_i = \mathrm{ResBlock}_{3P \to d_{\text{model}}}\bigl([\mathbf{p}_i; \mathbf{m}_i; \mathbf{r}_i]\bigr) \in \R^{d_{\text{model}}},
    \label{eq:method-projection}
\end{equation}
where $\mathrm{ResBlock}$ consists of a widened linear layer followed by a SiLU nonlinearity, a second linear layer, and a residual connection. Only the first linear layer widens; the transformer decoder, attention layers, output heads, and normalisation layers are unchanged.

\paragraph{Channel dropout.}
During training, each patch's preconditioning channel is independently zeroed with probability $p_{\text{drop}} = 0.1$; at inference the full preconditioning signal is always supplied. The primary motivation of channel dropout is to improve long horizon inference where we might have noisy preconditioning signals that are generated from the model's own autoregressive rollouts. Dropout increases the robustness of the model to missing or noisy preconditioning signals and therefore prevents catastrophic dependence or error accumulation. 


\paragraph{Parameter count.}
For Moirai-2 Small ($P = 16$, $d_{\text{model}} = 384$), widening the first linear layer from $32 \to 384$ (target + mask) to $48 \to 384$ (target + mask + preconditioning) adds $16 \cdot 384 = 6{,}144$ parameters; the residual path contributes another $\approx 6{,}000$. The total overhead is $\approx 12$K parameters, or $0.105\%$ of the $11.4$M baseline. No other parameters are added.

\begin{algorithm}[t]
\caption{Training with Polynomial Input Preconditioning}
\label{alg:poly-train}
\begin{algorithmic}[1]
\REQUIRE Packed patched sequence $\mathbf{p}_{1:N}$ with observation masks $\mathbf{m}_{1:N}$; prefix ratio $\alpha = 0.3$; mask ratio $\beta = 0.5$; Chebyshev coefficients $c_{1:d}$ (fixed); stride $s = P$; channel dropout rate $p_{\mathrm{drop}} = 0.1$; multi-token horizon $K = 4$.
\STATE \textbf{Partition:} first $\lfloor \alpha N \rfloor$ patches are context; remaining patches are prediction window.
\STATE \textbf{$z$-score:} compute $\mu, \sigma$ from observed values in the context patches. Set $\tilde{\mathbf{p}}_i \leftarrow (\mathbf{p}_i - \mu)/\sigma$ for all $i$.
\STATE \textbf{Mask:} for each prediction-window patch independently with probability $\beta$: zero out $\tilde{\mathbf{p}}_i$ and $\mathbf{m}_i$.
\FOR[Preconditioning residual]{$i = 1$ \TO $N$}
    \STATE $\mathbf{r}_i \leftarrow \sum_{k=1}^{d} c_k \cdot \tilde{\mathbf{p}}_{i-k}$ \COMMENT{Stride $s{=}P$ $\Rightarrow$ lag $k$ patches}
    \IF{$\mathrm{Bernoulli}(p_{\mathrm{drop}})$}
        \STATE $\mathbf{r}_i \leftarrow \mathbf{0}$ \COMMENT{Channel dropout}
    \ENDIF
\ENDFOR
\FOR[Embed]{$i = 1$ \TO $N$}
    \STATE $\mathbf{e}_i \leftarrow \mathrm{ResBlock}_{3P \to d_{\mathrm{model}}}([\tilde{\mathbf{p}}_i;\, \mathbf{m}_i;\, \mathbf{r}_i])$
\ENDFOR
\STATE $\mathbf{h}_{1:N} \leftarrow \mathrm{TransformerDecoder}(\mathbf{e}_{1:N})$ \COMMENT{Causal attention}
\STATE $\hat{\mathbf{y}}_{i}^{(q,k)} \leftarrow \mathrm{OutProj}(\mathbf{h}_i)$ for quantiles $q \in \mathcal{Q}$, horizons $k \in \{1,\ldots,K\}$.
\STATE $\mathcal{L} \leftarrow \frac{1}{|\mathcal{Q}|} \sum_{q,i,k} \rho_q\!\bigl(\tilde{\mathbf{p}}_{i+k} - \hat{\mathbf{y}}_i^{(q,k)}\bigr)$ over prediction-window positions.
\STATE Update all trainable parameters (ResBlock, Transformer, OutProj) via $\nabla \mathcal{L}$.
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[t]
\caption{Inference with Polynomial Input Preconditioning}
\label{alg:poly-infer}
\begin{algorithmic}[1]
\REQUIRE Observed history $x_{1:T}$; prediction length $H$; Chebyshev coefficients $c_{1:d}$ (fixed); stride $s = P$; multi-token horizon $K = 4$; quantile levels $\mathcal{Q} = \{0.1, \ldots, 0.9\}$.
\ENSURE Point forecasts $\hat{x}_{T+1:T+H}$.
\STATE \textbf{$z$-score:} compute $\mu, \sigma$ from $x_{1:T}$. Set $\tilde{x}_t \leftarrow (x_t - \mu)/\sigma$ for $t = 1, \ldots, T$.
\STATE \textbf{Patchify context:} $\tilde{\mathbf{p}}_i \leftarrow (\tilde{x}_{(i-1)P+1}, \ldots, \tilde{x}_{iP})$, $\mathbf{m}_i \leftarrow \mathbf{1}$ for $i = 1, \ldots, T/P$.
\STATE Initialise prediction patches: $\tilde{\mathbf{p}}_{T/P+j} \leftarrow \mathbf{0}$, $\mathbf{m}_{T/P+j} \leftarrow \mathbf{0}$ for $j = 1, \ldots, \lceil H/P \rceil$.
\STATE \textbf{Compute residual:} $\mathbf{r}_i \leftarrow \sum_{k=1}^{d} c_k \cdot \tilde{\mathbf{p}}_{i-k}$ for all patches $i$. \COMMENT{No dropout at inference}
\STATE \textbf{Embed, decode:} $\mathbf{e}_i \leftarrow \mathrm{ResBlock}([\tilde{\mathbf{p}}_i; \mathbf{m}_i; \mathbf{r}_i])$; $\mathbf{h}_{1:N} \leftarrow \mathrm{TransformerDecoder}(\mathbf{e}_{1:N})$.
\STATE \textbf{Read predictions:} from the last context patch $i^* = T/P$, extract $\hat{\mathbf{y}}_{i^*}^{(0.5, k)}$ for $k = 1, \ldots, K$. These are the median forecasts for the next $K$ patches ($KP$ time steps).
\IF[Direct prediction]{$H \leq KP$}
    \STATE \textbf{Return} $\hat{x}_{T+1:T+H} \leftarrow \sigma \cdot \hat{\mathbf{y}}^{(0.5)}_{1:H} + \mu$.
\ELSE[Autoregressive rollout]
    \STATE Fill predicted patches into context: $\tilde{\mathbf{p}}_{i^*+k} \leftarrow \hat{\mathbf{y}}_{i^*}^{(0.5,k)}$, $\mathbf{m}_{i^*+k} \leftarrow \mathbf{1}$.
    \STATE Recompute $\mathbf{r}$ for newly filled patches using predicted values.
    \STATE Repeat from \textbf{Embed, decode} with extended context until $H$ steps are covered.
\ENDIF
\end{algorithmic}
\end{algorithm}

Algorithms~\ref{alg:poly-train} and~\ref{alg:poly-infer} describe training and inference respectively. Three points are worth noting. First, because the stride $s$ equals the patch size $P$, the preconditioning residual $\mathbf{r}_i$ at patch $i$ depends on patches $i{-}1, i{-}2, \ldots, i{-}d$, making the computation a simple lag over patches rather than individual time steps. Second, the Chebyshev coefficients are fixed throughout; the only trainable parameters are the widened input projection and the existing transformer and output-projection weights. Third, during autoregressive rollout (Algorithm~\ref{alg:poly-infer}, line 11), the residual $\mathbf{r}$ for newly predicted patches is computed from the model's own median predictions, which may be noisy; the channel dropout applied during training (Algorithm~\ref{alg:poly-train}, line 8) regularises against this mismatch.

\subsection{Experimental Setup}

When we write $s = 16$ in ablation tables, this refers to experiments using the default $P = 16$ configuration. The four conditions of interest below share identical training data, optimizer, schedule, seeds, and evaluation pipeline; only the content of the auxiliary input channel differs. The ablations in Table~\ref{tab:ablations-main} align with the theoretical predictions of Section~\ref{sec:precondition} and pin down what is and is not load-bearing about \HD{}'s configuration.

\emph{(a) Filter degree.} We vary the preconditioning degree across $d \in \{2, \ldots, 8\}$ and running five seeds each. We note that $d = 4$ achieves the lowest mean MASE, but the MASE scores across all the degrees are roughly within one seed-to-seed standard deviation. In any case, we note that for every degree we get an improvement over the baseline. We use $d=4$ as a working choice. 

% but the curve is shallow: $d \in \{3, 5, 6, 8\}$ are within roughly one seed-to-seed standard deviation of $d = 4$, and only $d = 2$ (a single non-trivial lag) and $d = 7$ trail meaningfully. The seed-robust qualitative statement is ``every $d \geq 2$ improves over \Baseline{}''---which is what the spectral-compression argument predicts (Theorem~\ref{thm:chebyshev-minimax}, Corollary~\ref{cor:spectral-compression}). The flatness of the degree curve is consistent with the regret-bound tradeoff of Theorem~\ref{thm:usp-general}: the spectral-compression term saturates by $d \approx 3$ on capacity-limited patch transformers, after which additional taps offer diminishing benefit relative to growing coefficient-norm cost. We therefore treat $d=4$ as a working choice, not a uniquely optimal one.

\emph{(b) Stride.} Stride $s = P = 16$ is optimal, beating $s = 4, 8, 32$. This is the empirical signature of the cross-patch interpretation, as filter taps at lags $32$ and $64$ align exactly with patch boundaries (two and four patches back), so the preconditioning channel injects cross-patch structure at the natural unit of the architecture. We suspect that strides smaller than $P$ contaminate the residual with within-patch information that the per-patch projection already sees; strides much larger than $P$ over-shoot the receptive scales the model can use.

\emph{(c) Basis.} We compare four filter bases, all at $d = 4$, $s = 16$, $10\%$ dropout. To understand the ordering, we describe each basis and its explicit FIR coefficients for $d = 4$:

\begin{itemize}[leftmargin=1.5em, itemsep=0.3em, topsep=0.3em]
    \item \textbf{Monic Chebyshev} ($\tilde T_4(z) = z^4 - z^2 + 1/8$). The residual FIR is $r_t = -x_{t-2P} + \tfrac{1}{8} x_{t-4P}$---only two nonzero taps. The $L^\infty$ norm on $[-1,1]$ is $\sup_{[-1,1]} \abs{\tilde T_4} = 2^{-3} = 1/8 = 0.125$, the minimax-optimal value (Theorem~\ref{thm:chebyshev-minimax}).

    \item \textbf{Monic Legendre} ($\tilde P_4(z) = z^4 - \tfrac{6}{7}z^2 + \tfrac{3}{35}$). Defined via the Bonnet recurrence $(d{+}1)P_{d+1}(x) = (2d{+}1)x P_d(x) - d P_{d-1}(x)$, then dividing by the leading coefficient to obtain the monic form. The residual FIR is $r_t = -\tfrac{6}{7} x_{t-2P} + \tfrac{3}{35} x_{t-4P}$---the same sparsity pattern as Chebyshev but different coefficients. Legendre polynomials minimise the $L^2$ norm $(\int_{-1}^1 p(x)^2\, dx)^{1/2}$ among monic polynomials, but their $L^\infty$ norm is strictly larger: $\sup_{[-1,1]} \abs{\tilde P_4} = 8/35 \approx 0.229$, a $1.83\times$ gap versus Chebyshev (and $\approx 3.35\times$ in the squared term that enters the spectral-radius bound~\eqref{eq:rho-bound}).

    \item \textbf{Exponential moving average} (EMA). A one-sided low-pass filter with geometrically decaying weights $a_k \propto \alpha^k$, truncated to $d = 4$ taps. Not derived from a polynomial family. EMA provides some cross-patch mixing but has no minimax optimality property.

    \item \textbf{Finite difference}. The $d$-th order difference operator $p(z) = (z - 1)^d$. For $d = 4$: $r_t = -4 x_{t-P} + 6 x_{t-2P} - 4 x_{t-3P} + x_{t-4P}$---a heavy high-pass filter (the discrete fourth derivative). Its $L^\infty$ norm on $[-1,1]$ is $\sup_{[-1,1]} \abs{(z-1)^4} = 2^4 = 16$, amplifying rather than compressing spectral content. It has no spectral-attenuation guarantee.
\end{itemize}

The empirical ordering Chebyshev ($0.823$) $>$ EMA ($0.833$) $>$ Legendre ($0.840$) $>$ FiniteDiff ($0.857$) directly matches the $L^\infty$-norm prediction of the spectral-radius bound (Eq.~\eqref{eq:rho-bound}): the basis that minimises $\sup_{[-1,1]} \abs{p}$ wins, and the bases that provide no attenuation (finite differences) barely help. The Chebyshev--Legendre gap of $\sim 1.7$ percentage points is consistent with the $1.83\times$ theoretical sup-norm ratio, and the fact that all polynomial bases except finite differences beat \Baseline{} ($0.862$) confirms that cross-patch information injection is the load-bearing mechanism---the specific coefficients then determine how much benefit is extracted.

\emph{(d) Channel dropout.} We find that a $10\%$ dropout rate is optimal.

\subsection{Fixed vs.\ Learned Coefficients}
\label{sec:learned-fir}

A natural alternative to fixing $p$ to be the monic Chebyshev polynomial is to drop the polynomial structure entirely and let the model learn its own degree-$d$ coefficients $(c_1, \ldots, c_d)$ jointly with the rest of the network. This isolates the question: does the gain come from the \emph{specific} Chebyshev coefficients prescribed by Theorem~\ref{thm:chebyshev-minimax}, or merely from the existence of \emph{some} cross-patch lookback channel that the model can shape end-to-end? We trained two learned variants matching \HD{}'s architecture exactly except that the coefficients are now four trainable scalars: one variant initialised at the Chebyshev coefficients, the other at zero.

Three things are worth pulling out of Table~\ref{tab:learned-fir}.

\emph{(i) Either learned variant beats \Baseline{}.} A learnt preconditioner over multi-patch lookbacks captures cross-patch correlations the un-augmented Moirai-2 input projection misses. This rules out the strong null in which any auxiliary lookback channel hurts, and reinforces the cross-patch-bottleneck story: the auxiliary channel acts on the same temporal scales the architecture cannot reach within a single patch.

\emph{(ii) Initialisation determines the basin: Chebyshev-init stays near Chebyshev, zero-init drifts to a smoother low-pass.} The Chebyshev-initialised variant ends up close to its starting point: the dominant coefficient remains at lag $2s = 32$ with weight $\approx -1$, indicating that the loss landscape has a basin of attraction around the analytic solution. The zero-initialised variant converges to a qualitatively different solution: a smoother low-pass filter. 

Two distinct local minima of comparable quality is a sign that perhaps the basin is empirically pretty broad and various preconditioning filters that mix multi-patch information may be roughly as good. 

\emph{(iii) Fixed slightly outperforms either learned variant.} This is the empirical signal that the Chebyshev choice is meaningful as an inductive bias, and makes the model more sample-efficient.

\section{Discussion and Limitations}
\label{sec:discussion}
%==============================================================================

The main empirical claim that preconditioning can improve the performance of patch-based time series models is well-supported. For the from-scratch Moirai-2-Small-style architecture and training budget studied here, appending the fixed degree-four Chebyshev residual improves normalized MASE on GIFT-Eval and fev-bench. 

We believe that a plausible mechanism that accounts for this performance improvement is the input-side bottleneck present in the original Moirai-2 model, as the per-patch projection can only see one patch before attention. The residual channel provides information from 2 and 4 patches back, so the first embedding would already contain additional useful information. And the Chebyshev coefficients appear uniquely suited because they provide the minimax spectral advantage demonstrated in USP.

However, we also note some limitations of our work. In particular, our experiments do not provide any provable guarantees for applying preconditioning techniques to patch-based time series models. The mathematical background mainly serves as intuition for proposing a principled fixed filter (Chebyshev basis), which we then demonstrate empirically to work well. 

Moreover, we only demonstrate the benefit of preconditioning at the small ($11.4$M) Moirai-2 scale. Due to limited compute, we are not able to perform similarly comprehensive experiments at the larger ($46M$) base scale. Also, the method was tested primarily in a univariate channel setting, whereas multivariate extensions might require different considerations.

\section{Conclusion}
%==============================================================================
In this paper, we first discussed the mathematical background of online learning of linear dynamical systems and motivate the theory behind Universal Sequence Preconditioning. We provide expository work to obtain the $\tilde O(T^{-2/13})$ regret bound for USP applied with online convex regression. We then applied these techniques to patch-based universal time series forecasters using a minimal architectural intervention that incorporates the preconditioned signal as an auxiliary channel. Empirically, on the GIFT-Eval benchmark ($-2.9\%$ MASE at $10$K steps, growing to $-3.9\%$ at $100$K; $329/485$ paired wins) and independently on FEV-Bench ($-2.9\%$ MASE, $72/100$ wins), polynomial preconditioning consistently improves a capacity-limited Moirai-2 Small model. Capacity-matched controls confirmed that the polynomial content, and not the extra parameters, drives the gain. Overall, we have demonstrated the empirical effectiveness of USP in time series forecasting, and believe that many similar improvements can be obtained by incorporating USP signals into other deep learning models.

\subsection{Evaluation Protocol}

All evaluations use the GIFT-Eval benchmark \citep{vijay2024gifteval} with 97 dataset$\times$horizon configurations:
\begin{itemize}[leftmargin=1.5em]
    \item Context length: 4{,}000 tokens (maximum supported)
    \item Patch size: 32 for most frequencies, 8 for quarterly (following Moirai conventions)
    \item Batch size: 64
    \item Metric: Normalized MASE (each configuration's MASE divided by the seasonal naive baseline's MASE)
    \item Aggregation: geometric mean across 97 configurations
    \item Reported quantities: cross-seed mean and standard deviation, per-seed MASE, and per-configuration win counts (out of 97 per seed; $5 \times 97 = 485$ when pooled across seeds)
\end{itemize}

\subsection{Preconditioning Channel Implementation}

The Chebyshev polynomial $T_d(z)$ is expanded in the monomial basis to obtain FIR filter coefficients $c_0, \ldots, c_d$. The preconditioned signal for time step $t$ is
\[
    r_t \;=\; \left(\sum_{k=0}^{d} c_k \, x_{t-ks}\right) - x_t,
\]
computed over the full (normalized, zero-mean unit-variance) time series before patching. Unobserved time steps are masked to zero. During training with channel dropout $p$, each patch's preconditioning channel independently has probability $p$ of being replaced with zeros. At inference, the full preconditioning channel is always provided.

The preconditioning channel is concatenated with the target and mask channels before the input projection. The input projection is the only architectural change relative to Baseline; all downstream layers (transformer decoder, output projection) are identical.

\section{Polynomial Coefficients}
\label{app:poly-coeffs}
%==============================================================================

This appendix lists the explicit FIR coefficients for every polynomial basis discussed in this paper. Throughout, we use the convention of Section~\ref{sec:precondition}: writing a monic polynomial in descending powers as $p(z) = \sum_{i=0}^{d} c_i z^{d-i}$ with $c_0 = 1$, the residual filter applied to the input is $r_t = \sum_{k=1}^{d} c_k\, x_{t-ks}$ (the non-leading coefficients at stride $s$). The default stride is $s = P = 16$.

\paragraph{Monic Chebyshev (first kind), $\tilde T_d(x) = T_d(x)/2^{d-1}$.}
The classical Chebyshev polynomials satisfy the recurrence $T_0 = 1$, $T_1(x) = x$, $T_{d+1}(x) = 2x T_d(x) - T_{d-1}(x)$. After dividing by the leading coefficient $2^{d-1}$, the monic forms are obtained explicitly. The FIR coefficients $(c_1, \ldots, c_d)$ at stride $s$ are listed in Table~\ref{tab:cheb-coeffs}.

\begin{table}[h]

Effective lookback in time-step units is $ks$ for the $k$-th nonzero tap. Boldface marks the default $d = 4$ filter used by \HD{} and \texttt{d4}.}
\label{tab:cheb-coeffs}

\end{tabular}

For odd $d$, the monic Chebyshev polynomial $\tilde T_d(z)$ has zero coefficients at all even powers of $z$ (since $T_d(-z) = -T_d(z)$ for odd $d$). Writing $\tilde T_d(z) = z^d + c_1 z^{d-1} + \cdots + c_d$ in descending powers, the coefficients $c_k$ with even $k$ (i.e., at odd powers of $z$) vanish. Since the residual $r_t = \sum_{k=1}^d c_k x_{t-ks}$ sums over these coefficients, only the odd-indexed $c_k$ contribute, yielding nonzero taps at even multiples of $s$ only. For example, the $d=3$ residual at $s = 16$ is supported only at lag $2s = 32$; the $d=5$ residual at $s = 16$ is supported at lags $2s = 32$ and $4s = 64$.

\paragraph{Monic Chebyshev second kind, $\tilde U_d(x) = U_d(x)/2^d$.}
Defined by $U_0 = 1$, $U_1(x) = 2x$, $U_{d+1}(x) = 2x U_d(x) - U_{d-1}(x)$. The first few monic forms and corresponding residual FIRs at $s = 16$ are:
\[
\begin{aligned}
\tilde U_2(x) &= x^2 - \tfrac{1}{4}, &\quad r_t &= -\tfrac{1}{4} x_{t-32}, \\
\tilde U_3(x) &= x^3 - \tfrac{1}{2} x, &\quad r_t &= -\tfrac{1}{2} x_{t-32}, \\
\tilde U_4(x) &= x^4 - \tfrac{3}{4} x^2 + \tfrac{1}{16}, &\quad r_t &= -\tfrac{3}{4} x_{t-32} + \tfrac{1}{16} x_{t-64}.
\end{aligned}
\]
The $\ell_\infty$ norm of monic $\tilde U_d$ on $[-1,1]$ is $1/2^{d-1}$ \citep[Eq.~1.41]{mason2003chebyshev}, twice that of $\tilde T_d$, so the spectral attenuation predicted by the equioscillation argument is half as strong as for the first kind.

\paragraph{Monic Legendre, $\tilde P_d(x) = P_d(x) \cdot 2^d (d!)^2 / (2d)!$.}
Defined by Bonnet's recurrence $(d+1) P_{d+1}(x) = (2d+1) x P_d(x) - d P_{d-1}(x)$, $P_0 = 1$, $P_1(x) = x$. The leading coefficient of $P_d$ is $(2d)!/(2^d (d!)^2)$, so dividing by it yields the monic form. The first few monic Legendre polynomials and residual FIRs at $s = 16$ are:
\[
\begin{aligned}
\tilde P_2(x) &= x^2 - \tfrac{1}{3}, &\quad r_t &= -\tfrac{1}{3} x_{t-32}, \\
\tilde P_3(x) &= x^3 - \tfrac{3}{5} x, &\quad r_t &= -\tfrac{3}{5} x_{t-32}, \\
\tilde P_4(x) &= x^4 - \tfrac{6}{7} x^2 + \tfrac{3}{35}, &\quad r_t &= -\tfrac{6}{7} x_{t-32} + \tfrac{3}{35} x_{t-64}.
\end{aligned}
\]
The $\ell_\infty$ norm of monic $\tilde P_d$ on $[-1,1]$ is strictly larger than that of $\tilde T_d$ for all $d \geq 2$. The exact gap is given by the following short proposition, which restates the Chebyshev minimax theorem (Theorem~\ref{thm:chebyshev-minimax}) against the Legendre $L^2$ minimiser.

\begin{proposition}[Sup-norm gap between Chebyshev and Legendre]
\label{prop:cheb-vs-legendre}
For $d \geq 1$,
\[
    \frac{\sup_{[-1,1]} \abs{\tilde T_d}}{\sup_{[-1,1]} \abs{\tilde P_d}} \;=\; \frac{2 \binom{2d}{d}}{4^d} \;\xrightarrow[d \to \infty]{}\; \frac{2}{\sqrt{\pi d}}.
\]
The ratio equals $1$ at $d = 1$ and is strictly less than $1$ for all $d \geq 2$, with values $1, 3/4, 5/8, 35/64, 63/128, \ldots$ for $d = 1, 2, 3, 4, 5, \ldots$ respectively.
\end{proposition}

\begin{proof}
$\sup_{[-1,1]} \abs{\tilde T_d} = 2^{1-d}$ from Theorem~\ref{thm:chebyshev-minimax}. For Legendre: $\abs{P_d(x)} \leq 1$ on $[-1,1]$ with equality at $x = \pm 1$, so the monic $\tilde P_d = P_d / c_d$ with leading coefficient $c_d = (2d)!/(2^d (d!)^2)$ has sup norm $1/c_d = 2^d (d!)^2 / (2d)!$. Taking the ratio and applying Stirling's approximation gives $2 \binom{2d}{d}/4^d \to 2/\sqrt{\pi d}$.
\end{proof}

For $d = 4$, the monic Chebyshev attenuates a worst-case eigenvalue by $1/8 = 0.125$, while the monic Legendre attenuates by only $8/35 \approx 0.229$---a $1.83\times$ gap, or $\approx 3.35\times$ when squared. This explains the $\sim 1.9$-percentage-point Chebyshev advantage observed in the basis ablation (Table~\ref{tab:ablations-main}(c)). Legendre minimises the $\ell_2$ norm $\bigl(\int_{-1}^{1} p(x)^2 \, dx\bigr)^{1/2}$ rather than the $\ell_\infty$ norm; for the LDS spectral-radius bound (Eq.~\eqref{eq:rho-bound}), only the $\ell_\infty$ optimisation is the relevant one.

\paragraph{Other bases used in ablations.}
\begin{itemize}[leftmargin=1.5em]
    \item \textbf{Finite-difference} (degree $d$): $p(z) = (z - 1)^d$. For $d = 4$ at $s = 16$ this expands to $r_t = -4 x_{t-16} + 6 x_{t-32} - 4 x_{t-48} + x_{t-64}$. Heavy high-pass behaviour, no spectral-attenuation guarantee.
    \item \textbf{Exponential moving average} (EMA, half-life $\tau$): a one-sided low-pass filter with infinite impulse response coefficients $a_k = (1-\alpha) \alpha^k$, truncated to the window we use. Not polynomial-based.
    \item \textbf{Learned FIR} (no fixed basis): the model learns the $d = 4$ residual coefficients $(c_1, c_2, c_3, c_4) \in \R^4$ from a Gaussian initialisation jointly with the rest of the network. The learned solution depends on initialisation (Section~\ref{sec:learned-fir}, Table~\ref{tab:learned-fir}).
\end{itemize}
