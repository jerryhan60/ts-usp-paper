%=== COLUMN 1 ======================================================
\panel{1 — Problem: The Cross-Patch Bottleneck}{
{\large
Universal time series forecasters \emph{patch} the input into blocks of $P$ samples and embed each patch independently before any attention layer. A daily cycle in $15$-minute data spans $96$ steps $= 6$ patches---\textbf{attention alone must discover this}.

\vspace{8pt}
Smaller models, with fewer heads/layers, are most constrained. We inject cross-patch information \emph{before the first attention layer} via a fixed polynomial filter that preserves parameter count.

\vspace{6pt}
\includegraphics[width=\linewidth]{figures/fig_architecture.pdf}
}}

\panel{2 — Our Method: Hint Preconditioning}{
{\large
Compute the \textbf{hint channel}
\[
  h[t] \;=\; \sum_{k=0}^{d} a_k\, x[t - k s], \qquad s = P,
\]
and \emph{concatenate} it with the target and observation mask before the per-patch projection:
\[
  \mathbf{e}_i = \mathrm{ResBlock}_{3P \to d}([\mathbf{p}_i;\,\mathbf{m}_i;\,\tilde{\mathbf{h}}_i]).
\]
For $d = 4$, $s = 16$:
\[
  h[t] \;=\; x[t] - x[t{-}32] + \tfrac{1}{8}\, x[t{-}64].
\]
Only the first linear layer widens; the transformer encoder is unchanged.

\vspace{6pt}
\textbf{Hint dropout.} At training each patch's hint is zeroed with prob.\ $0.1$; at inference the full hint is provided.
}}

%=== COLUMN 2 ======================================================
\panel{3 — Why Chebyshev? (Theory)}{
{\large
Two classical results motivate the choice of filter coefficients.

\vspace{6pt}
\textbf{(i) Chebyshev minimax theorem.} Among all monic polynomials of degree $d$,
\[
  \tilde T_d \;=\; \arg\min_{q \text{ monic}} \;\max_{x \in [-1,1]} |q(x)|, \quad \|\tilde T_d\|_\infty = 2^{1-d}.
\]

\textbf{(ii) LDS preconditioning} (Marsden \& Hazan '25): applying $p(B)$ to an observation sequence $y_t = C A^t h_0$ is equivalent to multiplying the hidden dynamics by $p^*(A)$. The Chebyshev choice minimizes the worst-case spectral amplification.

\vspace{6pt}
\textbf{Frequency response (new).} For our filter at stride $s$ the magnitude
\[
  |H(\omega)|^2 = p\!\big(e^{-j 2\pi s\omega}\big)\,\overline{p\!\big(e^{-j 2\pi s\omega}\big)}
\]
is a real trigonometric polynomial in $\cos(2\pi s\omega)$, and attains its minimum value
\[
  |H(\omega_k)| \;=\; 2^{\,1 - d} \quad \text{at}\quad \omega_k = k/(2s), \; k \in \mathbb{Z}.
\]
}}

\panel{4 — Synthetic LDS: Theory Validation}{
{\large\centering
\includegraphics[width=\linewidth]{figures/synth_lds_spectrum.pdf}\par
\raggedright
Simulated LDS ($\rho = 0.99$, $m = 5$, stride $s = 3$).\\
Filter notches land at $\omega = 0, 1/6, 1/3$---\textbf{exactly where the theorem predicts}. Notch depth scales as $2^{2-2d}$, matching the Chebyshev minimax constant.
}}

\panel{6 — Statistical Methodology (CSML)}{
{\large
\begin{itemize}[leftmargin=1em]
\item Paired sign test, $485$ observations: $p = 2.9\times 10^{-15}$.
\item \textbf{Holm--Bonferroni} per-seed: $4$ of $5$ seeds individually significant at $\alpha = 0.05$.
\item \textbf{BCa 95\% bootstrap} ($B = 5{,}000$): relative improvement $\in [-3.71\%, -2.12\%]$; excludes zero.
\item Cohen's $d = +0.31$ (paired log-MASE); Cliff's $\delta = +0.36$.
\item Post-hoc power: $n = 65$ suffices for $80\%$ power; we have $485$ ($7.5\times$).
\end{itemize}
}}

{\LARGE\bfseries\color{accentGreen} Takeaways}

\begin{multicols}{3}
\large
\textbf{1. Theory meets practice.} Chebyshev preconditioning is \emph{minimax-optimal} for spectral compression; empirically, it beats all tested alternatives (EMA, Legendre, finite differences, duplicate, zero).

\columnbreak
\textbf{2. Capacity-efficient.} $0.11\%$ extra parameters yield $-2.9\%$ MASE and $-3.9\%$ at $100$K steps. No change to the transformer encoder.

\columnbreak
\textbf{3. Predictable from spectrum.} Datasets with more spectral mass in the filter's passband benefit more ($r = +0.10$, $39$ datasets)---mechanism consistent with Theorem~2.
\end{multicols}
